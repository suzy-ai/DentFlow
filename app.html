import { serve } from "https://deno.land/std@0.168.0/http/server.ts"; const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "Content-Type, Authorization", "Content-Type": "application/json", }; serve(async (req) => { // Handle CORS preflight if (req.method === "OPTIONS") { return new Response("ok", { headers: corsHeaders }); } if (req.method !== "POST") { return new Response(JSON.stringify({ error: "Method not allowed" }), { status: 405, headers: corsHeaders, }); } try { const { messages } = await req.json(); const apiKey = Deno.env.get("ANTHROPIC_API_KEY"); if (!apiKey) { return new Response(JSON.stringify({ error: "API key not configured." }), { status: 500, headers: corsHeaders, }); } const response = await fetch("https://api.anthropic.com/v1/messages", { method: "POST", headers: { "Content-Type": "application/json", "x-api-key": apiKey, "anthropic-version": "2023-06-01", }, body: JSON.stringify({ model: "claude-haiku-4-5-20251001", max_tokens: 512, system: "You are DentFlow, a dental office assistant in the United States. You must always respond in English. Never respond in any other language. You help dental staff with scheduling, billing, insurance, supplies, and office management. Keep answers short and practical.", messages: messages, }), }); const data = await response.json(); if (!response.ok) { return new Response(JSON.stringify({ error: data.error?.message || "API error" }), { status: response.status, headers: corsHeaders, }); } return new Response(JSON.stringify({ reply: data.content[0].text }), { status: 200, headers: corsHeaders, }); } catch (err) { return new Response(JSON.stringify({ error: "Server error: " + err.message }), { status: 500, headers: corsHeaders, }); } });
